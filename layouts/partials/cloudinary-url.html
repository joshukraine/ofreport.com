{{- $src := .src -}} {{- $preset := .preset -}} {{- $result := "" -}} {{- if $src -}} {{- /* Three source shapes are supported: 1. bare public ID -> expand to a full Cloudinary /upload/ URL 2. full Cloudinary URL -> inject transforms after /upload/ 3. any other remote http URL -> deliver via Cloudinary fetch (/image/fetch//) Shape 3 is how legacy WordPress images (still hosted on the old CloudFront distribution) get optimized + lightbox-ready without re-uploading them. */ -}} {{- $isRemote := false -}} {{- if hasPrefix $src "//" -}} {{- /* normalize protocol-relative URLs to https before anything else */ -}} {{- $src = printf "https:%s" $src -}} {{- end -}} {{- if not (hasPrefix $src "http") -}} {{- $src = printf "%s/image/upload/%s" site.Params.cloudinaryBase $src -}} {{- else if not (hasPrefix $src site.Params.cloudinaryBase) -}} {{- /* a full http URL that is not under our own Cloudinary base -> fetch */ -}} {{- $isRemote = true -}} {{- end -}} {{- /* Display presets use c_limit, not c_scale, so an intrinsically small image is delivered at its native size rather than upscaled (#135): a 300px legacy cover renders crisp-but-small instead of blurry-stretched. c_limit and c_scale are identical for the curated covers (all >= 2000px wide); the guard only matters if a small cover ever appears. og keeps c_fill — social cards need exact 1200x630 dimensions. */ -}} {{- $presets := dict "featured" "c_limit,f_auto,q_auto,w_740" "regular" "c_limit,f_auto,q_auto,w_610" "figure" "c_limit,f_auto,q_auto,w_1344,h_1344" "article" "c_scale,f_auto,q_auto:best" "hero" "c_limit,f_auto,q_auto:best,w_2000" "cover" "c_scale,f_auto,q_auto:best,w_200" "lightbox" "c_scale,f_auto,q_auto,w_1600" "og" "c_fill,f_auto,h_630,q_auto,w_1200" "rss" "c_scale,f_auto,q_auto,w_560" "bg" "c_scale,f_auto,q_auto,w_2000" -}} {{- $transforms := index $presets $preset -}} {{- if $isRemote -}} {{- /* Cloudinary fetch: the remote URL is appended raw after the transforms. */ -}} {{- if $transforms -}} {{- $result = printf "%s/image/fetch/%s/%s" site.Params.cloudinaryBase $transforms $src -}} {{- else -}} {{- $result = printf "%s/image/fetch/%s" site.Params.cloudinaryBase $src -}} {{- end -}} {{- else if $transforms -}} {{- $result = replace $src "/upload/" (printf "/upload/%s/" $transforms) -}} {{- else -}} {{- $result = $src -}} {{- end -}} {{- end -}} {{- return $result -}}
